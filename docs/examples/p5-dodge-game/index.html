'use client';

import { P5Canvas } from '@p5-wrapper/react';

export default function DodgeGame() {
  return (
    <P5Canvas
      sketch={(p) => {
        const PLAYER_W = 36;
        let playerX;
        let blocks = [];
        let spawnTimer = 0;
        let elapsed = 0;
        let alive = true;

        function reset() {
          playerX = p.width / 2;
          blocks = [];
          spawnTimer = 0;
          elapsed = 0;
          alive = true;
        }

        p.setup = () => {
          p.createCanvas(p.windowWidth, 400);
          reset();
        };

        p.draw = () => {
          p.background(15, 23, 42);

          if (alive) {
            playerX = p.constrain(p.mouseX, PLAYER_W / 2, p.width - PLAYER_W / 2);
            elapsed += p.deltaTime / 1000;

            const speed = 140 + elapsed * 12;
            spawnTimer -= p.deltaTime / 1000;
            if (spawnTimer <= 0) {
              blocks.push({ x: p.random(20, p.width - 20), y: -20, size: p.random(18, 32) });
              spawnTimer = Math.max(0.35, 0.9 - elapsed * 0.02);
            }

            for (const b of blocks) b.y += speed * (p.deltaTime / 1000);
            blocks = blocks.filter((b) => b.y < p.height + 40);

            for (const b of blocks) {
              const dx = Math.abs(b.x - playerX);
              const dy = Math.abs(b.y - (p.height - 30));
              if (dx < (b.size + PLAYER_W) / 2 && dy < (b.size + 16) / 2) {
                alive = false;
              }
            }
          }

          p.noStroke();
          p.fill(239, 68, 68);
          for (const b of blocks) p.rect(b.x - b.size / 2, b.y - b.size / 2, b.size, b.size, 4);

          p.fill(99, 102, 241);
          p.rect(playerX - PLAYER_W / 2, p.height - 38, PLAYER_W, 16, 4);

          p.fill(226, 232, 240);
          p.textSize(14);
          p.textAlign(p.LEFT, p.TOP);
          p.text(`Score: ${Math.floor(elapsed * 10)}`, 12, 12);

          if (!alive) {
            p.fill(15, 23, 42, 200);
            p.rect(0, 0, p.width, p.height);
            p.fill(226, 232, 240);
            p.textAlign(p.CENTER, p.CENTER);
            p.textSize(20);
            p.text('Game over — click to restart', p.width / 2, p.height / 2);
          }
        };

        p.mousePressed = () => { if (!alive) reset(); };
      }}
    />
  );
}

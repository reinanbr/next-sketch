'use client';

import { Canvas, useFrame } from '@react-three/fiber';
import { OrbitControls } from '@react-three/drei';
import { useRef } from 'react';
import * as THREE from 'three';

const PLANETS = [
  { radius: 3.2, size: 0.35, speed: 0.6, color: '#6366f1' },
  { radius: 4.6, size: 0.5, speed: 0.35, color: '#3b82f6' },
  { radius: 6.2, size: 0.28, speed: 0.22, color: '#ef4444' },
];

function Planet({ radius, size, speed, color }) {
  const pivot = useRef<THREE.Group>(null!);
  useFrame((_, dt) => { pivot.current.rotation.y += speed * dt; });
  return (
    <group ref={pivot}>
      <mesh position={[radius, 0, 0]}>
        <sphereGeometry args={[size, 24, 24]} />
        <meshStandardMaterial color={color} />
      </mesh>
    </group>
  );
}

export default function OrbitSimulation() {
  return (
    <Canvas style={{ width: '100%', height: 400 }} camera={{ position: [0, 6, 12], fov: 50 }}>
      <ambientLight intensity={0.3} />
      <pointLight position={[0, 0, 0]} intensity={80} color="#fde68a" />
      <mesh>
        <sphereGeometry args={[1.1, 32, 32]} />
        <meshBasicMaterial color="#fde68a" />
      </mesh>
      {PLANETS.map((p, i) => <Planet key={i} {...p} />)}
      <OrbitControls enableDamping />
    </Canvas>
  );
}

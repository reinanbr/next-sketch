'use client';

import { P5Canvas } from '@p5-wrapper/react';

const PARTICLE_COUNT = 500;

export default function FlowField() {
  return (
    <P5Canvas
      sketch={(p) => {
        let particles = [];
        let zoff = 0;

        p.setup = () => {
          p.createCanvas(p.windowWidth, 400);
          particles = Array.from({ length: PARTICLE_COUNT }, () => ({
            x: p.random(p.width),
            y: p.random(p.height),
          }));
          p.background(15, 23, 42);
        };

        p.draw = () => {
          p.noStroke();
          p.fill(15, 23, 42, 12);
          p.rect(0, 0, p.width, p.height);

          for (const pt of particles) {
            const angle = p.noise(pt.x * 0.005, pt.y * 0.005, zoff) * p.TWO_PI * 2;
            pt.x += Math.cos(angle) * 1.5;
            pt.y += Math.sin(angle) * 1.5;
            if (pt.x < 0) pt.x = p.width;
            if (pt.x > p.width) pt.x = 0;
            if (pt.y < 0) pt.y = p.height;
            if (pt.y > p.height) pt.y = 0;

            p.stroke(99, 102, 241, 160);
            p.point(pt.x, pt.y);
          }
          zoff += 0.003;
        };
      }}
    />
  );
}
